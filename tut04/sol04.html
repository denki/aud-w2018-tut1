1.3 Teilaufgabe c

Seien ρ(S) = ρ(A) = {(ba)ⁿb ∣ n ∈ ℕ}.  Zu zeigen: ⟦{ba}A⟧(ρ) = ρ(S).

Beweis:

⟦{ba}A⟧(ρ)
= ⟦{ba}⟧(ρ) ⋅ ⟦A⟧(ρ)
= (⟦ba⟧(ρ))* ⋅ ⟦A⟧(ρ)
= (⟦b⟧(ρ) ⋅ ⟦a⟧(ρ))* ⋅ ⟦A⟧(ρ)
= ({b} ⋅ {a})* ⋅ ⟦A⟧(ρ)
= ({b} ⋅ {a})* ⋅ ρ(A)
= ({b} ⋅ {a})* ⋅ {(ba)ⁿb ∣ n ∈ ℕ}
= {ba}* ⋅ {(ba)ⁿb ∣ n ∈ ℕ}
= {(ba)ᵐ ∣ m ∈ ℕ} ⋅ {(ba)ⁿb ∣ n ∈ ℕ}
= {(ba)ᵐ(ba)ⁿb ∣ m, n ∈ ℕ}
= {(ba)ᵏb ∣ m, n ∈ ℕ, k = m + n}
= {(ba)ᵏb ∣ k ∈ ℕ}
= {(ba)ⁿb ∣ n ∈ ℕ}
= ρ(S)

#include<stdio.h>

int main() {
  int n;

  printf("Bitte Zahl eingeben: ");
  scanf("%i", &n);
  printf("\n");

  for (int i = 1; i <= n; ++i) {
    for (int j = 1; j <= n; ++j)
      printf("%i\t", i * j);
    printf("\n");
  }

  return 0;
}

2.3 Teilaufgabe a

#include<stdio.h>

#define LEN 101

int main() {
  unsigned int sieve[LEN];

  sieve[0] = 1;
  sieve[1] = 1;

  for (int i = 2; i <= LEN - 1; ++i)
    sieve[i] = 0;

  for (int i = 2; i <= LEN - 1; ++i)
    if (sieve[i] == 0) {
      printf("%u ", i);
      for (int j = 2 * i; j <= LEN; j = j + i)
        sieve[j] = 1;
    }

  printf("\n");

  return 0;
}
